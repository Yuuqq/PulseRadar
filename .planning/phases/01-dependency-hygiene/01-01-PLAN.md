```python
def _create_remote_backend(self) -> Optional[StorageBackend]:
    """创建远程存储后端"""
    try:
        from trendradar.storage.remote import HAS_BOTO3
    except ImportError:
        HAS_BOTO3 = False

    if not HAS_BOTO3:
        raise ImportError(
            "S3 远程存储已配置，但未安装 boto3。\n"
            "S3 storage is configured but boto3 is not installed.\n"
            "Install with: pip install trendradar[s3]"
        )

    try:
        from trendradar.storage.remote import RemoteStorageBackend

        return RemoteStorageBackend(
            bucket_name=self.remote_config.get("bucket_name") or os.environ.get("S3_BUCKET_NAME", ""),
            access_key_id=self.remote_config.get("access_key_id") or os.environ.get("S3_ACCESS_KEY_ID", ""),
            secret_access_key=self.remote_config.get("secret_access_key") or os.environ.get("S3_SECRET_ACCESS_KEY", ""),
            endpoint_url=self.remote_config.get("endpoint_url") or os.environ.get("S3_ENDPOINT_URL", ""),
            region=self.remote_config.get("region") or os.environ.get("S3_REGION", ""),
            enable_txt=self.enable_txt,
            enable_html=self.enable_html,
            timezone=self.timezone,
        )
    except ImportError:
        raise
    except Exception as e:
        logger.error("远程后端初始化失败", error=str(e))
        return None
```

# coding=utf-8
"""
测试 boto3 可选依赖的缺失检测

验证当 S3 存储已配置但 boto3 未安装时，StorageManager 会产生
包含安装指令的明确错误信息。
"""

from __future__ import annotations

import importlib
import sys
from unittest.mock import patch

import pytest


def test_missing_boto3_raises_with_install_command():
    """配置 S3 存储但未安装 boto3 时，应抛出包含安装命令的 ImportError。"""
    # Save references to modules we'll remove
    saved_modules = {}
    modules_to_block = [
        "boto3", "botocore", "botocore.config",
        "botocore.exceptions", "botocore.session",
    ]

    for mod_name in list(sys.modules.keys()):
        if mod_name == "boto3" or mod_name.startswith("botocore"):
            saved_modules[mod_name] = sys.modules.pop(mod_name)

    # Also remove cached remote module so it re-evaluates HAS_BOTO3
    if "trendradar.storage.remote" in sys.modules:
        saved_modules["trendradar.storage.remote"] = sys.modules.pop(
            "trendradar.storage.remote"
        )

    # Block boto3/botocore from being imported
    blocker = {name: None for name in modules_to_block}

    try:
        with patch.dict("sys.modules", blocker):
            # Force reimport of remote module so HAS_BOTO3 = False
            if "trendradar.storage.remote" in sys.modules:
                del sys.modules["trendradar.storage.remote"]

            from trendradar.storage.manager import StorageManager

            sm = StorageManager(
                backend_type="remote",
                remote_config={
                    "bucket_name": "test-bucket",
                    "access_key_id": "AKID_TEST",
                    "secret_access_key": "SECRET_TEST",
                    "endpoint_url": "https://s3.example.com",
                },
            )
            with pytest.raises(ImportError, match=r"pip install trendradar\[s3\]"):
                sm.get_backend()
    finally:
        # Restore original modules
        for mod_name, mod in saved_modules.items():
            sys.modules[mod_name] = mod


def test_missing_boto3_error_is_bilingual():
    """错误信息应同时包含中文和英文说明。"""
    saved_modules = {}
    modules_to_block = [
        "boto3", "botocore", "botocore.config",
        "botocore.exceptions", "botocore.session",
    ]

    for mod_name in list(sys.modules.keys()):
        if mod_name == "boto3" or mod_name.startswith("botocore"):
            saved_modules[mod_name] = sys.modules.pop(mod_name)

    if "trendradar.storage.remote" in sys.modules:
        saved_modules["trendradar.storage.remote"] = sys.modules.pop(
            "trendradar.storage.remote"
        )

    blocker = {name: None for name in modules_to_block}

    try:
        with patch.dict("sys.modules", blocker):
            if "trendradar.storage.remote" in sys.modules:

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| pyproject.toml -> pip resolver | Dependency version ranges control what code is downloaded and executed |
| requirements.txt -> Docker build | Auto-generated file consumed by CI/CD without human review of each line |
| Error message -> user terminal | Error text with install command shown to user; low risk of info disclosure |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Tampering | pyproject.toml tenacity range `>=9.0,<10` | accept | Research confirmed tenacity is not imported anywhere in source code (zero code impact). Range is bounded with `<10` upper cap. Supply chain risk is identical to all other range-specified deps in this project. |
| T-01-02 | Tampering | requirements.txt auto-generated by uv | mitigate | Task 1 verifies generated content (structlog present, boto3 absent, no exact pin). Header documents the regen command for auditability. Generated from the same pyproject.toml source of truth. |
| T-01-03 | Information Disclosure | Error message in fail-fast check | accept | Message contains only the package name and install command (`pip install trendradar[s3]`). No internal paths, no config values, no credentials exposed. Acceptable for a startup error. |
| T-01-04 | Denial of Service | Removing boto3 from core deps breaks existing users | mitigate | Docker images (Plan 02) explicitly install boto3. Error message is directive ("Install with: pip install trendradar[s3]"). CHANGELOG documents the change (Plan 02). Existing `HAS_BOTO3` guard in remote.py preserved as defense-in-depth. |
| T-01-05 | Elevation of Privilege | boto3 as optional extra reduces default attack surface | accept | This is a POSITIVE security change: ~100MB of boto3+botocore code is no longer installed for users who don't need S3. Reduced attack surface. |
