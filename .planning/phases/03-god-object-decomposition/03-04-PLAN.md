**What the new NewsAnalyzer looks like (D-06):**

```python
class NewsAnalyzer:
    """Thin facade preserving the original public interface."""

    def __init__(self, config: Optional[Dict] = None, update_info: Optional[Dict] = None):
        if config is None:
            logger.info("Loading config")
            config = load_config()
        logger.info("Config loaded", version=__version__,
                     platforms=len(config['PLATFORMS']),
                     timezone=config.get('TIMEZONE', DEFAULT_TIMEZONE))

        self.ctx = AppContext(config)
        self.update_info = update_info

        # Proxy setup (keep existing logic from _setup_proxy)
        self.proxy_url = None
        is_github_actions = os.environ.get("GITHUB_ACTIONS") == "true"
        if not is_github_actions and self.ctx.config["USE_PROXY"]:
            self.proxy_url = self.ctx.config["DEFAULT_PROXY"]

        # Storage init (keep existing logic from _init_storage_manager)
        env_retention = os.environ.get("STORAGE_RETENTION_DAYS", "").strip()
        if env_retention:
            self.ctx.config["STORAGE"]["RETENTION_DAYS"] = int(env_retention)
        self.storage_manager = self.ctx.get_storage_manager()

        # Create orchestrators
        self.crawl_coordinator = CrawlCoordinator(self.ctx, proxy_url=self.proxy_url)
        self.analysis_engine = AnalysisEngine(
            self.ctx, update_info=self.update_info, proxy_url=self.proxy_url
        )

    def run(self) -> None:
        """Execute the analysis pipeline."""
        try:
            self._log_startup()
            crawl_output = self.crawl_coordinator.crawl_all()
            analysis_output = self.analysis_engine.analyze(crawl_output)
        except Exception as e:
            logger.error("Pipeline error", error=str(e))
            if self.ctx.config.get("DEBUG", False):
                raise
        finally:
            self.ctx.cleanup()

    def _log_startup(self) -> None:
        """Log startup info (time, crawler status, notification status, mode)."""
        now = self.ctx.get_time()
        logger.info("Current time", time=now.strftime('%Y-%m-%d %H:%M:%S'))
        # ... keep existing logging from _initialize_and_check_config lines 368-386
```
